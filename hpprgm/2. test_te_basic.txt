#pragma mode( separator(.,;) integer(h32) )

EXPORT test_te_basic()
BEGIN
  LOCAL res, cl, cc;
  LOCAL p_cnt, f_cnt, ypos;
  LOCAL ln, lcount, ok;
  LOCAL grn, red, gry, blu;
  LOCAL heb_word, rus_word, mix_line;
  LOCAL orig_len, new_len, final_len;
  LOCAL idx, doc200, ln1;

  grn := RGB(0,160,0);
  red := RGB(200,0,0);
  gry := RGB(150,150,150);
  blu := RGB(0,80,200);

  p_cnt := 0;
  f_cnt := 0;
  ypos := 0;

  RECT_P(0,0,319,219,RGB(255,255,255),RGB(255,255,255));
  TEXTOUT_P("TE Basic Test running...", 2, ypos, 1, blu);
  ypos := ypos + 14;
  LINE_P(0, ypos-2, 319, ypos-2, gry);

  // Scenario 01: Empty document init
  res := TE_Init();
  cl := res(1);
  cc := res(2);
  ok := (TE_LineCount() == 1) AND (cl == 1) AND (cc == 1) AND (TE_GetLine(1) == "");
  IF ok THEN
    TEXTOUT_P("S01 PASS init: 1 line, cur={1,1}", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S01 FAIL init empty doc", 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 02: Insert chars into empty line
  res := TE_Init();
  cl := res(1);
  cc := res(2);
  res := TE_InsertChar(cl, cc, "A"); cl := res(1); cc := res(2);
  res := TE_InsertChar(cl, cc, "B"); cl := res(1); cc := res(2);
  res := TE_InsertChar(cl, cc, "C"); cl := res(1); cc := res(2);
  ln := TE_GetLine(1);
  ok := (ln == "ABC") AND (cl == 1) AND (cc == 4);
  IF ok THEN
    TEXTOUT_P("S02 PASS insert ABC, col=4", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S02 FAIL got: "+ln+" col="+STRING(cc), 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 03: Mid-line insertion
  res := TE_InitFrom({"ABEF"});
  cl := res(1); cc := res(2);
  res := TE_MoveRight(cl, cc); cl := res(1); cc := res(2);
  res := TE_MoveRight(cl, cc); cl := res(1); cc := res(2);
  res := TE_InsertText(cl, cc, "CD"); cl := res(1); cc := res(2);
  ln := TE_GetLine(1);
  ok := (ln == "ABCDEF") AND (cc == 5);
  IF ok THEN
    TEXTOUT_P("S03 PASS mid-insert: ABCDEF", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S03 FAIL got: "+ln, 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 04: Newline split
  res := TE_InitFrom({"ABCD"});
  cl := res(1); cc := 3;
  res := TE_Newline(cl, cc); cl := res(1); cc := res(2);
  ok := (TE_LineCount() == 2) AND (TE_GetLine(1) == "AB") AND (TE_GetLine(2) == "CD") AND (cl == 2) AND (cc == 1);
  IF ok THEN
    TEXTOUT_P("S04 PASS newline: AB|CD, cur={2,1}", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S04 FAIL L1="+TE_GetLine(1)+" L2="+TE_GetLine(2), 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 05: Backspace inside line
  res := TE_InitFrom({"ABCDE"});
  cl := 1; cc := 4;
  res := TE_Backspace(cl, cc); cl := res(1); cc := res(2);
  ln := TE_GetLine(1);
  ok := (ln == "ABDE") AND (cc == 3);
  IF ok THEN
    TEXTOUT_P("S05 PASS backspace: ABDE, col=3", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S05 FAIL got: "+ln+" col="+STRING(cc), 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 06: Backspace at line start -> merge
  res := TE_InitFrom({"Hello","World"});
  cl := 2; cc := 1;
  res := TE_Backspace(cl, cc); cl := res(1); cc := res(2);
  ln := TE_GetLine(1);
  ok := (TE_LineCount() == 1) AND (ln == "HelloWorld") AND (cl == 1) AND (cc == 6);
  IF ok THEN
    TEXTOUT_P("S06 PASS bs@start: HelloWorld, col=6", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S06 FAIL got: "+ln+" lines="+STRING(TE_LineCount()), 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 07: Delete inside line
  res := TE_InitFrom({"ABCDE"});
  cl := 1; cc := 3;
  res := TE_Delete(cl, cc); cl := res(1); cc := res(2);
  ln := TE_GetLine(1);
  ok := (ln == "ABDE") AND (cc == 3);
  IF ok THEN
    TEXTOUT_P("S07 PASS delete: ABDE, col=3", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S07 FAIL got: "+ln, 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 08: Delete at line end -> merge
  res := TE_InitFrom({"AB","CD"});
  cl := 1; cc := 3;
  res := TE_Delete(cl, cc); cl := res(1); cc := res(2);
  ln := TE_GetLine(1);
  ok := (TE_LineCount() == 1) AND (ln == "ABCD") AND (cl == 1) AND (cc == 3);
  IF ok THEN
    TEXTOUT_P("S08 PASS del@end: ABCD, lines=1", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S08 FAIL got: "+ln+" lines="+STRING(TE_LineCount()), 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 09: Move across line boundaries
  res := TE_InitFrom({"AB","CD"});
  cl := 1; cc := 3;
  res := TE_MoveRight(cl, cc); cl := res(1); cc := res(2);
  ok := (cl == 2) AND (cc == 1);
  res := TE_MoveLeft(cl, cc); cl := res(1); cc := res(2);
  ok := ok AND (cl == 1) AND (cc == 3);
  IF ok THEN
    TEXTOUT_P("S09 PASS cross-line move", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S09 FAIL cur={"+STRING(cl)+","+STRING(cc)+"}", 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 10: Up/down with column clamping
  res := TE_InitFrom({"ABCDE","XY"});
  cl := 1; cc := 5;
  res := TE_MoveDown(cl, cc); cl := res(1); cc := res(2);
  ok := (cl == 2) AND (cc == 3);
  res := TE_MoveUp(cl, cc); cl := res(1); cc := res(2);
  ok := ok AND (cl == 1) AND (cc == 3);
  IF ok THEN
    TEXTOUT_P("S10 PASS up/down col clamp", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S10 FAIL cur={"+STRING(cl)+","+STRING(cc)+"}", 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 11: Hebrew/Russian/mixed logical storage
  heb_word := CHAR(1513)+CHAR(1500)+CHAR(1493)+CHAR(1502);
  rus_word := CHAR(1087)+CHAR(1088)+CHAR(1080)+CHAR(1074)+CHAR(1077)+CHAR(1090);
  mix_line := heb_word + " 123 " + rus_word;

  res := TE_InitFrom({heb_word, rus_word, mix_line});
  orig_len := SIZE(TE_GetLine(1));
  res := TE_InsertChar(1, 1, CHAR(1488));
  new_len := SIZE(TE_GetLine(1));
  res := TE_Backspace(1, 2);
  final_len := SIZE(TE_GetLine(1));
  ok := (new_len == orig_len + 1) AND (final_len == orig_len);
  ok := ok AND (SIZE(TE_GetLine(3)) == SIZE(mix_line));
  IF ok THEN
    TEXTOUT_P("S11 PASS HEB/RUS storage intact", 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S11 FAIL len="+STRING(final_len)+" exp="+STRING(orig_len), 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 12;

  // Scenario 12: 200-line stability smoke
  doc200 := {};
  FOR idx FROM 1 TO 200 DO
    doc200 := CONCAT(doc200, {"line"+STRING(idx)});
  END;
  res := TE_InitFrom(doc200);
  res := TE_InsertChar(1, 1, "X");
  res := TE_InsertChar(100, 1, "Y");
  res := TE_InsertChar(200, 1, "Z");
  res := TE_Newline(100, 1); cl := res(1); cc := res(2);
  lcount := TE_LineCount();
  ln1 := TE_GetLine(1);
  ok := (lcount == 201) AND (INSTRING(ln1, "X") == 1);
  IF ok THEN
    TEXTOUT_P("S12 PASS 200-line stable, lines="+STRING(lcount), 2, ypos, 1, grn);
    p_cnt := p_cnt + 1;
  ELSE
    TEXTOUT_P("S12 FAIL lines="+STRING(lcount)+" L1="+ln1, 2, ypos, 1, red);
    f_cnt := f_cnt + 1;
  END;
  ypos := ypos + 14;

  LINE_P(0, ypos-2, 319, ypos-2, gry);
  IF f_cnt == 0 THEN
    TEXTOUT_P("ALL PASS "+STRING(p_cnt)+"/12", 2, ypos, 2, grn);
  ELSE
    TEXTOUT_P(STRING(p_cnt)+"/12 PASS   "+STRING(f_cnt)+" FAIL", 2, ypos, 2, red);
  END;

  FREEZE;
END;
