#pragma mode( separator(.,;) integer(h32) )

// ================================================================
// Renderer MVP Test Harness  (Task 5)
//
// REQUIRES: renderer, hebrew_font_bitmap, text_engine modules loaded.
// ENTRY:    test_renderer_basic()
//
// 8 visual scenarios, each waits for a key press to advance.
// All Hebrew strings built via CHAR() -- no literal Hebrew in source.
//
// SCENARIO SUMMARY:
//   S1  Empty document           -- cursor on empty line
//   S2  LTR line                 -- cursor at start / middle / end
//   S3  RTL Hebrew line          -- cursor at right / middle / left
//   S4  Multi-line mixed         -- LTR and RTL lines interleaved
//   S5  Final forms              -- words ending in sofit variants
//   S6  Mixed-content fallback   -- Hebrew + digits -> placeholder boxes
//   S7  Cursor on empty line     -- empty line between content lines
//   S8  Viewport slice           -- R_VIEW_TOP = 5, 10-line document
// ================================================================


// ----------------------------------------------------------------
// S1: Empty document
// Expected: blank editor area, blue cursor at top-left (LTR cc=1 -> x=4)
// ----------------------------------------------------------------
EXPORT rend__scen1()
BEGIN
  LOCAL ev;
  TE_Init();
  REND_SetViewTop(1);
  REND_DrawStatus("S1: Empty doc  [key->next]");
  REND_DrawDoc(1, 1);
  ev := WAIT(-1);
END;


// ----------------------------------------------------------------
// S2: LTR line  "Hello World"  (len=11)
// S2a cursor at col 1  (left edge)
// S2b cursor at col 6  (after "Hello", before " World")
// S2c cursor at col 12 (after last char)
// Expected: text rendered LTR from x=4, cursor visible at each pos
// ----------------------------------------------------------------
EXPORT rend__scen2()
BEGIN
  LOCAL ev;
  TE_InitFrom({"Hello World"});
  REND_SetViewTop(1);
  REND_DrawStatus("S2a: LTR cursor start  [key]");
  REND_DrawDoc(1, 1);
  ev := WAIT(-1);
  REND_DrawStatus("S2b: LTR cursor middle  [key]");
  REND_DrawDoc(1, 6);
  ev := WAIT(-1);
  REND_DrawStatus("S2c: LTR cursor end  [key]");
  REND_DrawDoc(1, 12);
  ev := WAIT(-1);
END;


// ----------------------------------------------------------------
// S3: RTL Hebrew line
// hline = shin(1513) lamed(1500) vav(1493) mem-sofit(1501)  len=4
// S3a cursor col 1 -> rightmost visual position
// S3b cursor col 3 -> between char 2 and 3
// S3c cursor col 5 -> leftmost (after all 4 chars)
// Expected: 4 Hebrew glyphs from right side, cursor visible
// ----------------------------------------------------------------
EXPORT rend__scen3()
BEGIN
  LOCAL ev, hline;
  hline := CHAR(1513)+CHAR(1500)+CHAR(1493)+CHAR(1501);
  TE_InitFrom({hline});
  REND_SetViewTop(1);
  REND_DrawStatus("S3a: RTL cursor right  [key]");
  REND_DrawDoc(1, 1);
  ev := WAIT(-1);
  REND_DrawStatus("S3b: RTL cursor middle  [key]");
  REND_DrawDoc(1, 3);
  ev := WAIT(-1);
  REND_DrawStatus("S3c: RTL cursor left  [key]");
  REND_DrawDoc(1, 5);
  ev := WAIT(-1);
END;


// ----------------------------------------------------------------
// S4: Multi-line mixed (6 lines: alternating LTR and RTL)
// h1 = shalom (shin lamed vav mem-sofit)
// h2 = bereshit (bet resh alef shin yod tav)
// h3 = emet (alef mem tav)
// S4a: cursor on line 2 (Hebrew), col 1
// S4b: cursor on line 4 (Hebrew), col 3
// Expected: LTR lines on left, RTL lines on right side
// ----------------------------------------------------------------
EXPORT rend__scen4()
BEGIN
  LOCAL ev, h1, h2, h3, doc;
  h1 := CHAR(1513)+CHAR(1500)+CHAR(1493)+CHAR(1501);
  h2 := CHAR(1489)+CHAR(1512)+CHAR(1488)+CHAR(1513)+CHAR(1497)+CHAR(1514);
  h3 := CHAR(1488)+CHAR(1502)+CHAR(1514);
  doc := {"Line 1 LTR", h1, "Line 3 LTR", h2, "Line 5 LTR", h3};
  TE_InitFrom(doc);
  REND_SetViewTop(1);
  REND_DrawStatus("S4a: mixed, cursor line 2  [key]");
  REND_DrawDoc(2, 1);
  ev := WAIT(-1);
  REND_DrawStatus("S4b: mixed, cursor line 4  [key]");
  REND_DrawDoc(4, 3);
  ev := WAIT(-1);
END;


// ----------------------------------------------------------------
// S5: Final forms
// Each line is a Hebrew word ending in a sofit (final form):
//   L1: mem(1502)+lamed(1500)+kaf-sofit(1498)   -- melech (king)
//   L2: yod(1497)+mem-sofit(1501)               -- yam (sea)
//   L3: ayin(1506)+nun-sofit(1503)              -- "an" (eye-nun)
//   L4: alef(1488)+pe-sofit(1507)               -- "af" (alef-pe)
//   L5: alef(1488)+tsadi-sofit(1509)            -- "ets" (alef-tsadi)
// Expected: final forms clearly distinct from base forms
// ----------------------------------------------------------------
EXPORT rend__scen5()
BEGIN
  LOCAL ev, lk, lm, ln, lp, lts, doc;
  lk  := CHAR(1502)+CHAR(1500)+CHAR(1498);
  lm  := CHAR(1497)+CHAR(1501);
  ln  := CHAR(1506)+CHAR(1503);
  lp  := CHAR(1488)+CHAR(1507);
  lts := CHAR(1488)+CHAR(1509);
  doc := {lk, lm, ln, lp, lts};
  TE_InitFrom(doc);
  REND_SetViewTop(1);
  REND_DrawStatus("S5: Final forms  [key]");
  REND_DrawDoc(1, 1);
  ev := WAIT(-1);
END;


// ----------------------------------------------------------------
// S6: Mixed-content fallback
// hline = shin(1513)+lamed(1500)+"123"+vav(1493)+mem-sofit(1501)
// Line direction -> RTL (contains Hebrew)
// "123" chars -> gray placeholder boxes in RTL line
// Expected: 2 Hebrew glyphs, 3 gray boxes, 2 Hebrew glyphs (right to left)
// ----------------------------------------------------------------
EXPORT rend__scen6()
BEGIN
  LOCAL ev, hline;
  hline := CHAR(1513)+CHAR(1500)+"123"+CHAR(1493)+CHAR(1501);
  TE_InitFrom({hline});
  REND_SetViewTop(1);
  REND_DrawStatus("S6: RTL+digits fallback  [key]");
  REND_DrawDoc(1, 1);
  ev := WAIT(-1);
END;


// ----------------------------------------------------------------
// S7: Cursor on empty line between content lines
// doc = {heb_line, "", heb_line}  (3 lines)
// cursor at (2, 1) -- empty middle line
// Expected: cursor on blank line, surrounding Hebrew lines visible
// ----------------------------------------------------------------
EXPORT rend__scen7()
BEGIN
  LOCAL ev, h1, doc;
  h1  := CHAR(1513)+CHAR(1500)+CHAR(1493)+CHAR(1501);
  doc := {h1, "", h1};
  TE_InitFrom(doc);
  REND_SetViewTop(1);
  REND_DrawStatus("S7: Cursor on empty line  [key]");
  REND_DrawDoc(2, 1);
  ev := WAIT(-1);
END;


// ----------------------------------------------------------------
// S8: Viewport slice (R_VIEW_TOP = 5, 10-line document)
// Lines 1-4: not visible
// Lines 5-9: visible in viewport
// Cursor at line 7 (3rd visible row)
// Expected: lines 5-9 shown, line 7 highlighted, lines 1-4 NOT shown
// ----------------------------------------------------------------
EXPORT rend__scen8()
BEGIN
  LOCAL ev, h1, doc;
  h1  := CHAR(1513)+CHAR(1500)+CHAR(1493)+CHAR(1501);
  doc := {"Line 1", "Line 2", "Line 3", h1, "Line 5", "Line 6", h1, "Line 8", "Line 9", "Line 10"};
  TE_InitFrom(doc);
  REND_SetViewTop(5);
  REND_DrawStatus("S8: Viewport from line 5  [key]");
  REND_DrawDoc(7, 1);
  ev := WAIT(-1);
END;


// ----------------------------------------------------------------
// Main entry point
// Calls REND_Init() then runs all 8 scenarios in order.
// ----------------------------------------------------------------
EXPORT test_renderer_basic()
BEGIN
  REND_Init();
  rend__scen1();
  rend__scen2();
  rend__scen3();
  rend__scen4();
  rend__scen5();
  rend__scen6();
  rend__scen7();
  rend__scen8();
END;
