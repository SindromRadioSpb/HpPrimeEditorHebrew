// ================================================================
// Visual test v2: Hebrew Bitmap Font (14x20, 1-bit, 27 glyphs)
// PREREQUISITES: hebrew_font_bitmap.hpprgm must be loaded first.
// HOW TO RUN:   test_hfb_vis() from Home screen.
//
// Pages (tap/key to advance):
//   Page 1: S1 Grid of all 27 glyphs + S2 Alignment strips
//   Page 2: S3 Confusable pair review (11 pairs/triples)
//   Page 3: S4 Reading strips + S5 Base vs final-form comparison
// ================================================================

// ----------------------------------------------------------------
// Page 1: Glyph grid + alignment strips
// ----------------------------------------------------------------
EXPORT hfb__page1(blk, gry, blu)
BEGIN
  LOCAL idx, px, py, ok;
  LOCAL s3_ok;

  RECT_P(0, 0, 319, 239, RGB(255,255,255), RGB(255,255,255));
  TEXTOUT_P("HFB v2  27 Hebrew Glyphs  [Tap->pairs]", 2, 1, 1, blu);
  LINE_P(G0, 0, 12, 319, 12, gry);
  TEXTOUT_P("S1: Glyph grid", 2, 14, 1, blk);

  // Row A: glyphs 1-9
  FOR idx FROM 1 TO 9 DO
    px := (idx - 1) * 18 + 1;
    py := 26;
    RECT_P(px-1, py-1, px+14, py+20, gry, RGB(255,255,255));
    HFB_BlitGlyph(idx, px, py);
    TEXTOUT_P(STRING(idx), px+2, py+21, 1, blk);
  END;

  // Row B: glyphs 10-18
  FOR idx FROM 10 TO 18 DO
    px := (idx - 10) * 18 + 1;
    py := 60;
    RECT_P(px-1, py-1, px+14, py+20, gry, RGB(255,255,255));
    HFB_BlitGlyph(idx, px, py);
    TEXTOUT_P(STRING(idx), px, py+21, 1, blk);
  END;

  // Row C: glyphs 19-27
  FOR idx FROM 19 TO 27 DO
    px := (idx - 19) * 18 + 1;
    py := 94;
    RECT_P(px-1, py-1, px+14, py+20, gry, RGB(255,255,255));
    HFB_BlitGlyph(idx, px, py);
    TEXTOUT_P(STRING(idx), px, py+21, 1, blk);
  END;

  LINE_P(G0, 0, 126, 319, 126, gry);

  // S2a: base alignment strip (1-22 tight, 22x14=308px)
  TEXTOUT_P("S2a: Base 1-22 tight:", 2, 128, 1, blk);
  FOR idx FROM 1 TO 22 DO
    px := (idx - 1) * 14 + 2;
    HFB_BlitGlyph(idx, px, 140);
  END;

  // S2b: final forms strip (23-27)
  TEXTOUT_P("S2b: Finals 23-27:", 2, 163, 1, blk);
  FOR idx FROM 23 TO 27 DO
    px := (idx - 23) * 20 + 2;
    RECT_P(px-1, 174, px+14, 195, gry, RGB(255,255,255));
    HFB_BlitGlyph(idx, px, 175);
    TEXTOUT_P(STRING(idx), px, 196, 1, blk);
  END;

  // S3 mapping test (inline, compact)
  LINE_P(G0, 0, 210, 319, 210, gry);
  TEXTOUT_P("S3 CodeToIdx: ", 2, 212, 1, blk);
  s3_ok := 1;
  IF HFB_CodeToIdx(1488) <> 1  THEN s3_ok := 0; END;
  IF HFB_CodeToIdx(1514) <> 22 THEN s3_ok := 0; END;
  IF HFB_CodeToIdx(1498) <> 23 THEN s3_ok := 0; END;
  IF HFB_CodeToIdx(1501) <> 24 THEN s3_ok := 0; END;
  IF HFB_CodeToIdx(1503) <> 25 THEN s3_ok := 0; END;
  IF HFB_CodeToIdx(1507) <> 26 THEN s3_ok := 0; END;
  IF HFB_CodeToIdx(1509) <> 27 THEN s3_ok := 0; END;
  IF HFB_CodeToIdx(9999) <> 0  THEN s3_ok := 0; END;
  IF s3_ok THEN
    TEXTOUT_P("PASS - all 27 map correctly", 82, 212, 1, RGB(0,140,0));
  ELSE
    TEXTOUT_P("FAIL - check mapping table", 82, 212, 1, RGB(200,0,0));
  END;

  TEXTOUT_P("Glyphs fill rows 7-17 (14px body). Tap for pairs.", 2, 226, 1, gry);
END;

// ----------------------------------------------------------------
// Main entry point
// FREEZE replaced by WAIT(-1): more reliable blocking across
// all HP Prime contexts (confirmed working in Spike03).
// Each page waits for any key or touch before advancing.
// ----------------------------------------------------------------
EXPORT test_hfb_vis()
BEGIN
  LOCAL blk, gry, blu, ev;

  blk := RGB(0, 0, 0);
  gry := RGB(160, 160, 160);
  blu := RGB(0, 80, 200);

  HFB_Init();

  hfb__page1(blk, gry, blu);
  ev := WAIT(-1);

  hfb__page2(blk, gry, blu);
  ev := WAIT(-1);

  hfb__page3(blk, gry, blu);
  ev := WAIT(-1);
END;
